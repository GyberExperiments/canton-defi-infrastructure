**Было:**
```bash
PASSWORD="hardcoded"  # ❌ Опасно
```

**Стало:**
```bash
# Пароль из аргумента или env
if [ -n "$1" ]; then
    PASSWORD="$1"
elif [ -n "$ADMIN_PASSWORD" ]; then
    PASSWORD="$ADMIN_PASSWORD"
else
    echo "Пароль не указан!"
    exit 1
fi
```
